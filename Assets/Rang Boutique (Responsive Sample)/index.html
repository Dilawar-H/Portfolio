The Fix — Pure CSS Media Queries

/* Mobile Navigation Fix */
.nav-links {
  display: flex;
  gap: 1.5rem;
}

@media (max-width: 768px) {
  .nav-links {
    display: none; /* hidden on mobile */
  }
  .hamburger {
    display: block;
  }

  /* Grid: 4 col → 2 col */
  .product-grid {
    grid-template-columns: repeat(2, 1fr);
  }

  /* Hero: side-by-side → stacked */
  .hero {
    flex-direction: column;
  }
}
